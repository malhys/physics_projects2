\section{Dipôles en série et en dérivation}


\begin{frame}

	\begin{alertblock}{}
		
		\begin{itemize}
			\item Dans un \kw{circuit série}, la valeur de l'intensité du courant est la même en tout point du circuit, quel que soit l'ordre des dipôles : c'est la \kw{loi d'unicité de l'intensité}.
		\end{itemize}
		
			\begin{center}
				\includegraphics[scale=0.35]{../img/schema2}
			
				\vspace*{-0.5cm}
					\begin{equation*}
					I = I_1 = I_2
					\end{equation*}
		\end{center}
	\end{alertblock}
\end{frame}


\begin{frame}

\begin{alertblock}{}
	\begin{itemize}
		\item Dans un circuit comportant des \kw{dérivations}, l'intensité du courant dans la branche principale est égale à la \kw{somme des intensités} des courants dans les \kw{branches dérivées}.
		
		\begin{center}
			\includegraphics[scale=0.3]{../img/schema3}
			
			\vspace*{-0.5cm}
			\begin{equation*}
			I = I_1 + I_2
			\end{equation*}
		\end{center}
	
		\item Dans un circuit en dérivation, un \kw{n\oe ud} est le point de jonction d'au moins trois branches.
	\end{itemize}
	
\end{alertblock}
\end{frame}
