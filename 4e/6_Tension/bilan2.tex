\begin{mybilan}
	\begin{itemize}
		\item Un dipôle est prévu pour fonctionner avec une tension précise, c'est sa \kw{tension nominale}.\pause
		
		\item Il y a \kw{adaptation} du dipôle au générateur lorsque la tension électrique à ses bornes est \kw{proche de sa tension nominale}. \pause
		
		\item Il y a \kw{sous-tension} si la tension est inférieure à la tension nominale. Si elle est supérieure, il y a \kw{surtension}, le dipôle risque d'être détérioré. 
	\end{itemize}
	
	
\end{mybilan}
