\begin{mybilan}
	\begin{itemize}
		\item La tension électrique $U$ est mesurée en \kw{volt} (symbole $V$). %On utilise un \kw{voltmètre}.\pause
		
		\item Pour mesurer une tension, on branche un \kw{voltmètre en dérivation} aux bornes d'un dipôle. (voir fiche méthode 4 page 105).\pause
		
		\item La tension aux bornes d'un fil ou d'un interrupteur fermé est nulle.
		
		
	\end{itemize}

\end{mybilan}
