\begin{mybilan}
	
	\begin{itemize}

		\item Dans un circuit en série, la tension aux bornes du générateur est égale à la somme des tensions aux bornes des récepteurs : c'est la \kw{loi d'additivité des tensions}. (voir partie 1 du bilan p 86)\pause
			
			
		\item Dans un circuit comportant des \kw{dérivations}, la \kw{tension est la même} entre les bornes des dipôles branchés en \kw{dérivation}.(voir partie 2 du bilan p 86)
		
		
		
	\end{itemize}
\end{mybilan}
