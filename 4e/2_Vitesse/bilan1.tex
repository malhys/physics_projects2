\begin{mybilan}
	\begin{itemize}
		\item Un mouvement est \kw{uniforme} si la valeur de la vitesse est constante ; \kw{accéléré} si si cette valeur augmente et \kw{ralenti} si elle diminue. 
		\item Si le mouvement est uniforme, la relation \kw{$ v = \dfrac{d}{t} $}, permet de relier la vitesse de l'objet, la distance parcourue et la durée du parcours avec :
			\begin{itemize}
				\item d : distance parcourue en mètre (m)
				\item t :durée du trajet en seconde (s)
				\item v : vitesse en mètre par seconde (m/s).
			\end{itemize}
	\end{itemize}
\end{mybilan}
