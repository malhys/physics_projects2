\section{Mesure d'une intensité électrique}


\begin{mybilan}
	L'unité de mesure de l'\kw{intensité} électrique est l' \kw{ampère} (symbole $A$). 
	
	Un \kw{ampèremètre} permet de mesurer l'intensité du courant, il se branche \kw{en série} dans le circuit.

\section{Dipôles en série et en dérivation}


\begin{mybilan}
	\begin{itemize}
		\item Dans un \kw{circuit série}, la valeur de l'intensité du courant est la même en tout point du circuit, quel que soit l'ordre des dipôles : c'est la \kw{loi d'unicité de l'intensité}.
			
			
			\twoCol{
				\begin{center}
					\includegraphics[scale=0.5]{img/schema2}
				\end{center}
				
			\kw{\begin{center}
				\begin{equation*}
					I = I_1 = I_2
				\end{equation*}
			\end{center}}}
		
	
		\item Dans un circuit comportant des \kw{dérivations}, l'intensité du courant dans la branche principale est égale à la \kw{somme des intensités} des courants dans les \kw{branches dérivées}.
		
		
		
		\twoCol{
			\begin{center}
				\includegraphics[scale=0.5]{img/schema3}
			\end{center}
			
			\kw{\begin{center}
					\begin{equation*}
						I = I_1 + I_2
					\end{equation*}
		\end{center}}}
	
		\item Dans un circuit en dérivation, un \kw{n\oe ud} est le point de jonction d'au moins trois branches.
	\end{itemize}
\end{mybilan}
