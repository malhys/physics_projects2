\begin{mybilan}
	\begin{itemize}
		\item Un gaz peut être présent dans l'eau, \kw{naturellement} ou \kw{artificiellement}. Il est \kw{dissous}. 
		
		\item Le gaz est le \kw{soluté}, l'eau est le \kw{solvant}.
		
		\item La \kw{solubilité} d'un gaz est la masse maximale de gaz qu'il est possible de dissoudre dans un litre de solution.
	\end{itemize}
\end{mybilan}
