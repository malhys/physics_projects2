\begin{mybilan}
	\begin{itemize}
		\item Pour extraire un gaz dissous, on peut chauffer ou agiter l'eau qui le contient.
		\item La récupération du gaz se fait par \kw{déplacement d'eau}. 
		
		\item Le gaz dissous dans les boissons gazeuses est le dioxyde de carbone. Pour l'identifier, on utilise l'\kw{eau de chaux}. Elle <<se trouble>> en présence de dioxyde de carbone, il y a formation d'un \kw{précipité blanc}.
	\end{itemize}

\end{mybilan}
