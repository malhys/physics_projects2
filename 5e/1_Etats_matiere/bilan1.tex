\begin{mybilan}
	\begin{itemize}
		\item La matière est constituée d'\kw{atomes} qui peuvent se regrouper pour former des \kw{molécules}. 
		\item Une \kw{espèce chimique} est un ensemble d'atomes ou de molécules identiques.
		\item Un \kw{corps pur} est composé d'\kw{une seule espèce chimique}.
		\item Un mélange est composé de \kw{plusieurs espèces chimiques} différentes. 
	\end{itemize}
\end{mybilan}
