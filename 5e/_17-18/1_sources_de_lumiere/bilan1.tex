\begin{mybilan}
	\begin{itemize}
		\item Le soleil, les lampes et les vers luisants produisent la lumière qu'ils émettent : ce sont des \kw{sources primaires}.
		\item La Lune et tous les objets éclairés renvoient la lumière qu'ils reçoivent dans toutes les directions : ce sont des \kw{objets diffusants}.
	\end{itemize}
\end{mybilan}
