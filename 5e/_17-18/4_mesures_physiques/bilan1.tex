\begin{mybilan}
	\begin{itemize}
		\item Le \kw{volume} représente l'espace occupé par une substance.
		\item Une \kw{éprouvette graduée} permet de mesurer le volume d'un liquide.
		\item Pour mesurer correctement un volume à l'aide d'une éprouvette graduée, on la pose sur une surface plane et on positionne \kw{les yeux au niveau de la surface du liquide}. On lit la graduation \kw{à la base du ménisque}.
	\end{itemize}
\end{mybilan}
