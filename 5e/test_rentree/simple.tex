\begin{question}{rang}    
  En s'éloignant du Soleil, la Terre est la~?
  \begin{reponses}

    \mauvaise{Première planète}
    \mauvaise{Cinquième planète}
  \end{reponses}

\begin{question}{nature}    
  La Terre est une planète~?

  \end{reponses}
\end{question}
