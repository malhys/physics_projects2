\begin{mybilan}
	\begin{itemize}
		\item La Lune est éclairée par le Soleil et \kw{diffuse} une partie de la lumière reçue vers la Terre. La partie de la Lune visible depuis la Terre s'appelle une \kw{phase}.
		\item Les \kw{phases de la Lune} évoluent au cours de la rotation de la Lune autour de la Terre, sur une période appelée \kw{lunaison}.

	\end{itemize}
\end{mybilan}
