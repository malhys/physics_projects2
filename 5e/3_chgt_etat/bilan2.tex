\begin{mybilan}
	\begin{itemize}
		\item Au moment du changement d'état d'un corps pur, sa \kw{température ne change pas}, on observe un \kw{palier de température} sur la courbe d'évolution de la température en fonction du temps.
		\item Quand un mélange change d'état, sa température varie mais il n'y a pas de palier.
		\item Pendant \kw{sa fusion ou sa vaporisation} un corps \kw{reçoit de l'énergie}.
		\item Pendant \kw{sa solidification ou sa liquéfaction}, un corps \kw{perds de l'énergie}.
	\end{itemize}


\end{mybilan}
