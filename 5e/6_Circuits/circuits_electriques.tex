\section{Court-circuit et sécurité}

\begin{mybilan}
	\begin{itemize}
		\item Lorsque ses \kw{bornes sont reliées ensemble} par un fil, un dipôle est en \kw{court-circuit}.
		\item Un court-circuit d'une source d'énergie présente un \kw{danger d'incendie et de destruction de la source d'énergie}
		
		\item Pour protéger une installation on utilise des coupe-circuit comme des fusibles ou des disjoncteurs. Ils ouvrent le circuit si le courant est trop intense.
	\end{itemize}
\end{mybilan}
