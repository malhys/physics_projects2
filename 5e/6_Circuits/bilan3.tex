\begin{mybilan}
	
	Pour <<dessiner>> un circuit, il a été convenu d'une même représentation utilisée par tous.\pause
	
	\begin{itemize}
		\item Chaque élément d'un circuit est représenté par son \kw{symbole normalisé}.\pause
		\item On dit que l'on représente le circuit par un \kw{schéma électrique} (voir fiche méthode 5 page 106). 
	\end{itemize}
\end{mybilan}
