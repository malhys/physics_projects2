\begin{mybilan}
	\begin{itemize}
		\item L'unité de volume du système international est le \kw{mètre cube} ($m^3$). L'unité de volume la plus utilisée pour un liquide est le \kw{litre} ($L$).
		\item Lorsqu'on transvase un liquide, son volume ne change pas.
		\item Un litre et un décimètre cube représentent le même volume : \kw{1 $L =$ 1 $dm^3$}. Si on divise ces volumes par \num{1000}, l'égalité est toujours vérifiée : 1$mL$ = 1 $cm^3$
	\end{itemize}
\end{mybilan}
