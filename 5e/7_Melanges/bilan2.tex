\begin{mybilan}
	
	\begin{itemize}
		\item Un solide se \kw{dissout} dans un liquide, lorsque le mélange des deux est homogène.\pause
		 
		 
		 \item Le solide (appelé \kw{soluté}) est \kw{soluble} dans le liquide. Le liquide est le \kw{solvant}.\pause
		 
		 \item Un mélange obtenu par dissolution est une \kw{solution}.\pause
		 
		 \item La masse d'une solution est égale à la somme des masses du soluté et du solvant.\pause
		 
		 
	\end{itemize}


\end{mybilan}
