\begin{mybilan}
	\begin{itemize}
		\item \`A une température donnée, on ne peut dissoudre qu'une masse limitée de solide dans un volume donné de liquide.\pause
		
		\item La masse maximale de liquide que l'on peut dissoudre dans un litre d'eau est la \kw{solubilité}.\pause
		
		\item Lorsqu'on ne peut plus dissoudre de soluté, la solution est \kw{saturée}
	\end{itemize}
	
\end{mybilan}
