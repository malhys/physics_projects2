\begin{mybilan}
	\begin{itemize}
		\item Si on ne peu pas distinguer les composants d'un mélange à l'\oe il nu, il est \kw{homogène}. Sinon il est \kw{hétérogène}.\pause
		
		\item Si le mélange de deux liquides est homogène, ils sont \kw{miscibles}. Si ce mélange est hétérogène, ils sont \kw{non miscibles}.
	\end{itemize}
\end{mybilan}
