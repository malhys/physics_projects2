\begin{mybilan}
	\begin{itemize}
		\item Dans un \kw{circuit série} les dipôles sont reliés entre eux pour former \kw{une seule boucle};\pause
		
		\item Dans un \kw{circuit en dérivation} les dipôles sont reliés pour former \kw{plusieurs boucles}. Chaque boucle est indépendante des autres.

	\end{itemize}
\end{mybilan}
