\begin{mybilan}
	\begin{itemize}
		\item L'ensemble des positions prises par un objet au cours de son mouvement forme sa \kw{trajectoire}.
		
		\item Si la trajectoire décrit une ligne droite, le mouvement est \kw{rectiligne}.
		
		\item Si la trajectoire décrit un cercle, le mouvement est \kw{circulaire}.
		
		\item Sinon il est \kw{curviligne}.
	\end{itemize}
\end{mybilan}
