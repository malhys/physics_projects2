\begin{mybilan}
	\begin{itemize}
		\item Le \kw{pH d'une solution} (potentiel hydrogène) est une grandeur sans unité qui \kw{varie entre 0 et 14}.
		\item Si le pH est :
			\begin{itemize}
				\item \kw{inférieur à 7}, la solution est \kw{acide};
				\item \kw{égal à 7}, elle est \kw{neutre};
				\item \kw{supérieur à 7}, elle est basique.
			\end{itemize}
	\end{itemize}
\end{mybilan}
