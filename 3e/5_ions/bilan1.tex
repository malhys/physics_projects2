\begin{mybilan}
	\begin{itemize}
		\item Un atome (ou groupe d'atomes) peut perdre des électrons ou en gagne. Il possède alors une \kw{charge électrique}, c'est un \kw{ion}.\pause
		
		\item Si un atome perd un ou plusieurs électrons, il est chargé positivement, c'est un \kw{ion positif} (un cation).\pause
		
		\item S'il en gagne, il est chargé négativement, c'est un \kw{ion négatif} (un anion). \pause
		
		\item La \kw{formule chimique} d'un ion indique le \kw{nombre d'électrons} gagnés ou perdus.
	\end{itemize}

\end{mybilan}
