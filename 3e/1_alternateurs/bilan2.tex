\begin{mybilan}
	\begin{itemize}
		\item Un alternateur de bicyclette  se compose d'une bobine et d'un \kw{aimant} relié à un galet qui permet de le faire tourner.
		\item L'alternateur reçoit de l'\kw{énergie mécanique} lorsque l'on fait tourner le galet. Il convertit cette \kw{énergie mécanique} en \kw{énergie électrique} lorsque l'aimant tourne devant la bobine.
		\item On traduit les conversions énergétiques par un \kw{diagramme d'énergie}.

	\end{itemize}
\end{mybilan}
