\begin{mybilan}
	\begin{itemize}
		\item Lorsque l'on déplace un \kw{aimant} au voisinage d'une \kw{bobine} (ou une bobine au voisinage d'un aimant), on produit une tension qui est positive ou négative selon le sens du mouvement.
		\item Si l'on enchaîne les deux mouvements, on obtient une tension variable aux bornes de la bobine.
	\end{itemize}
\end{mybilan}
