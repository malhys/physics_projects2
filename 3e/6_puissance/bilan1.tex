\begin{mybilan}
	\begin{itemize}
		\item La \kw{fiche signalétique} d'un appareil électrique indique sa tension nominale (en volts, $V$) et sa puissance de fonctionnement (en watt, $W$).\pause
		\item On utilise des \kw{coupes-circuits} (fusibles, disjoncteurs) pour protéger les appareils électriques des \kw{surintensités}. Ils ouvrent le circuit si l'intensité du courant dépasse une certaine valeur.
	\end{itemize}

\end{mybilan}
