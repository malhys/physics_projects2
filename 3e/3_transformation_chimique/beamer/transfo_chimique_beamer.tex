\section{Conservation de la masse lors d'une transformation chimique}


%\begin{frame}
%	\input{../act2}
%\end{frame}
\begin{frame}

	\begin{alertblock}{}
		\begin{itemize}
			\item Une transformation chimique fait intervenir des \kw{réactifs} et forme des \kw{produits}.
			\item \kw{Loi de conservation de la masse de Lavoisier} : lors d'une transformation chimique, la masse des réactifs consommés est égale à celle des produits formés. \pause
		\end{itemize}
		  
		  \vspace*{-1cm}
		  
		  \begin{LARGE}
			  	
			  \begin{center}
			  	\begin{equation*}
			  		m_{r\acute{e}actifs \: consomm\acute{e}s} = m_{produits \: form\acute{e}s} 
			  	\end{equation*}
			  \end{center}
		  \end{LARGE}
	
		
	\end{alertblock}
\end{frame}


\section{Modélisation d'une transformation chimique}
\begin{frame}
	\begin{alertblock}{}
		\begin{itemize}
			\item Lors d'une transformation chimique, les atomes des réactifs se redistribuent pour former des produits.\pause
			\item L'\kw{équation de réaction} utilise les formules chimiques des réactifs et des produits et précise le sens dans lequel la transformation a lieu.\pause
			\item Le \kw{nombre} de chaque sorte d'atomes est \kw{le même} dans les réactifs et dans les produits. \pause
		\end{itemize}
	\end{alertblock}


	\begin{exampleblock}{Exemple : Combustion du méthane}
%	Combustion du méthane :
	
	%	\vspace*{1cm}
	
	\begin{center}
		\ce{CH4 + 2O2 -> CO2 + 2H2O}\pause
	\end{center}
	
	Une molécule de méthane réagit avec deux molécules de dioxygène pour donner une molécule de dioxyde de carbone et deux molécules d'eau.
	\end{exampleblock}

\end{frame}
