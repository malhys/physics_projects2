\begin{mybilan}
	\begin{itemize}
		
		\item Une \kw{action de contact} ne peut être exercée qu'entre deux corps en contact l'un avec l'autre.
		
		\item Si les deux corps ne sont pas en contact, c'est une \kw{action à distance}.
		
		\item Un \kw{diagramme objets-interactions} est utilisé pour représenter les actions exercées sur un objet.
		
		\item Une action exercée sur un corps entraine sa \kw{mise en mouvement} ou une \kw{modification} de son mouvement.
	\end{itemize}
\end{mybilan}
