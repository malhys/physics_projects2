\begin{mybilan}
	\begin{itemize}
		\item Une action s'exerçant sur un corps entraîne une \kw{modification de son mouvement ou une mise en mouvement}.
		\item Deux corps sont en \kw{interaction} si le mouvement de l'un dépend de la présence de l'autre. 
		\item Une \kw{action de contact} ne peut exister qu'entre deux corps en contact l'un avec l'autre.
		\item Une action entre deux corps est une \kw{action à distance} lorsqu'il n'y a pas de contact entre eux. 
	\end{itemize}
\end{mybilan}
